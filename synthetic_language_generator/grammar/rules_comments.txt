S -> N V N    # субъект выполняет действие с объектом
S -> N V      # простое предложение: субъект + действие
N -> Adj N    # определение перед существительным
N -> Num N    # количественное определение
N -> много N  # множественное число через "много"
N -> NOUN     # базовое существительное
Adj -> ADJ    # выбор прилагательного из словаря
Num -> NUM    # выбор числительного из словаря
V -> VERB     # выбор глагола из словаря
